	EmpNo numeric(3), 
	EmpName varchar(10) not null, 
	Gender varchar(1) not null, 
	Salary numeric(6) not null, 
	Address varchar(10) not null, 
	DNo numeric(3),  
	primary key(EmpNo) , 
	check (Gender in ('M', 'F')));
